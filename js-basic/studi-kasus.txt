const produk = [

  { id: 2, nama: "Mouse",    harga: 150000,  stok: 20, kategori: "Elektronik" },

   → tampilkan produk termahal
   → gunakan reduce

// Manajemen kontak

Data awal:
const kontak = [
  { id: 1, nama: "Budi",  telepon: "081234567890", grup: "Keluarga" },
  { id: 2, nama: "Siti",  telepon: "082345678901", grup: "Kerja"    },
  { id: 3, nama: "Zaq",   telepon: "083456789012", grup: "Teman"    },
  { id: 4, nama: "Andi",  telepon: "084567890123", grup: "Kerja"    },
  { id: 5, nama: "Rudi",  telepon: "085678901234", grup: "Keluarga" },
  { id: 6, nama: "Dewi",  telepon: "086789012345", grup: "Teman"    }
]

Buat 6 fungsi:

1. tampilkanSemua()
   → tampilkan semua kontak bernomor urut
   → gunakan destructuring

2. cariKontak(keyword)
   → cari berdasarkan nama
   → tampilkan semua hasil yang cocok
   → kalau tidak ketemu → tampilkan pesan error

3. filterGrup(grup)
   → tampilkan semua kontak berdasarkan grup
   → contoh: filterGrup("Kerja")

4. tambahKontak(nama, telepon, grup)
   → tambah kontak baru ke array
   → id otomatis = id terakhir + 1
   → tampilkan pesan sukses

5. hapusKontak(id)
   → hapus kontak berdasarkan id
   → tampilkan nama kontak yang dihapus
   → kalau id tidak ada → tampilkan pesan error

6. ringkasan()
   → tampilkan total kontak
   → tampilkan jumlah kontak per grup
   → contoh output:
      Total kontak: 6
      Keluarga: 2
      Kerja: 2
      Teman: 2
